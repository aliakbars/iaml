from sklearn.model_selection import GridSearchCV, train_test_split
from sklearn.preprocessing import LabelEncoder

le = LabelEncoder()

y = df['survived']

X['sex'] = le.fit_transform(X['sex'])
X['alone'] = le.fit_transform(X['alone'])
X = pd.concat([X, pd.get_dummies(X.embarked)], axis=1)
del X['embarked']

X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.33, random_state=RANDOM_STATE)

Hyperparameter Tuning¶

In [9]:
# Kode diadaptasi dari http://ahmedbesbes.com/how-to-score-08134-in-titanic-kaggle-challenge.html

clf = DecisionTreeClassifier()

parameter_grid = {
    'max_depth' : [4,5,6,7,8],
    'criterion': ['gini','entropy']
}

grid_search = GridSearchCV(clf,
                           scoring='accuracy',
                           param_grid=parameter_grid,
                           cv=10)

grid_search.fit(X_train, y_train)

print('Best accuracy: {:.2%}'.format(grid_search.best_score_))
print('Best parameters: {}'.format(grid_search.best_params_))

ROC Curve¶

In [11]:
from sklearn.metrics import roc_curve, roc_auc_score

clfs = [
    DecisionTreeClassifier,
    RandomForestClassifier,
    GradientBoostingClassifier
]

fig, ax = plt.subplots()
fig.set_size_inches(7,7)

for clf in clfs:
    c = clf()
    c.fit(X_train, y_train)
    score = c.predict_proba(X_test)
    fpr, tpr, thresholds = roc_curve(y_test, score[:,1])
    auc = roc_auc_score(y_test, score[:,1])
    ax.plot(fpr*100, tpr*100, label='{} (AUC = {:.2f})'.format(clf.__name__, auc*100))

ax.set_xlabel('False Positive Rate')
ax.set_ylabel('True Positive Rate')
ax.plot([0,100], [0,100], label='Dummy (AUC = 50.00)', linestyle='--')
ax.legend(loc=4)
plt.show()

Dummy classifier dalam kasus ini dihasilkan dengan memprediksi data uji berdasarkan probabilitas kelas prior dari data latih.
